## input 让输入的密码以*显示。
```
<input type="text">
$(document).ready(function () {
    var str = '';
    $("input").keyup(function() {
        var value = $(this).val();
        str += value.substr(value.length - 1, 1); //获取最后一个字符放到str,因为除了最后一个字符 其他的已经为*
        $(this).val(value.replace(/./g, '*')) //输入框内容全部变为*
    })
})
```
